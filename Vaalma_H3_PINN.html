Physics-informed neural network (PINN) for Heat Equation¶

Problem¶

We are going to assume we have a uniform bar of length $L$, where one end is placed on the origin and the other at $x = L$.

Here the change in temperature is the partial differential equation \begin{equation} u_t = a^2 u_{xx},\quad 0 < x < L,\ t > 0, \end{equation} where $a$ is a constant.

This is a homogeneous initial boundary value problem \begin{cases} u_t = a^2 u_{xx},\quad 0 < x < L,\ t > 0,\\[1em] u(0,t) = 0,\quad u(L,t)=0,\quad t > 0,\\[1em] u(x,0) = \varphi(x), \quad 0\leq x\leq L. \end{cases}

This can be solved by the standard Fourier method, which we will not be going into.

\begin{align} u(x,t) &= \sum_{n=1}^{\infty} \alpha_n e^{-n^2 \pi^2 a^2 t / L^2}\sin\frac{n\pi x}{L} \\ \alpha_n &= \frac{2}{L} \int_0^L \varphi(x)\sin\frac{n\pi x}{L}dx \end{align}

We can take $\varphi$ as a function that reiles on the length of the rod \begin{equation} \varphi(x) = x(x^2 -3Lx+2L^2). \end{equation}

The parameters $a$ and $L$ are going to be 1 for now.

# Parameters
L = 1  # Length of the domain
a = 1  # Thermal diffusivity
n_terms = 10  # Number of terms in the Fourier series

# Initial condition function (for example, a sine wave)
def phi(x,L):
    f = x*(x**2-3*L*x+2*L**2)
    return f

# Fourier series approximation
def fourier_series(x, t, n_terms, L, a):
    u = np.zeros_like(x)
    for n in range(1, n_terms + 1):
        a_n = 2 / L * np.trapz(phi(x,L) * np.sin(n * np.pi * x / L), x)  # Fourier coefficient
        u += a_n * np.sin(n * np.pi * x / L) * np.exp((-a**2 * (n * np.pi )**2 * t)/ L)
    return u

# Define the spatial domain and time
x = np.linspace(0, L, 100)
time_steps = np.linspace(0, 0.45, 10)  # Time steps from t=0 to t=2, with 5 intervals

# Plot the Fourier series at different times
plt.figure(figsize=(8, 6))
for t in time_steps:
    u = fourier_series(x, t, n_terms, L, a)
    plt.plot(x, u, label=f't = {t:.2f}')

# Plot the exact initial condition
plt.plot(x, phi(x,L), 'k--', label='Initial condition (phi(x))')

# Labeling the plot
plt.title(f"Fourier Series Approximation to Heat Equation Over Time")
plt.xlabel('x')
plt.ylabel('u(x,0)')
plt.legend()
plt.grid(True)
plt.show()

def exact_solution(a, L, x, t):
    u = ((12*L**3)/torch.pi**3)*(torch.exp(-torch.pi**2*a**2*t/L**2)*torch.sin(torch.pi*x/L) +
                             (1/8)*torch.exp(-4**torch.pi**2*a**2*t/L**2)*torch.sin(2*torch.pi*x/L) + 
                             (1/27)*torch.exp(-9*torch.pi**2*a**2*t/L**2)*torch.sin(3*torch.pi*x/L)+ 
                             (1/64)*torch.exp(-16*torch.pi**2*a**2*t/L**2)*torch.sin(4*torch.pi*x/L)+
                             (1/125)*torch.exp(-25*torch.pi**2*a**2*t/L**2)*torch.sin(5*torch.pi*x/L))
    return u

class FCN(nn.Module):
    "Defines a fully-connected network in PyTorch"
    def __init__(self, N_INPUT, N_OUTPUT, N_HIDDEN, N_LAYERS):
        super().__init__()
        activation = nn.Tanh
        self.fcs = nn.Sequential(*[
                        nn.Linear(N_INPUT, N_HIDDEN),
                        activation()])
        self.fch = nn.Sequential(*[
                        nn.Sequential(*[
                            nn.Linear(N_HIDDEN, N_HIDDEN),
                            activation()]) for _ in range(N_LAYERS-1)])
        self.fce = nn.Linear(N_HIDDEN, N_OUTPUT)
    def forward(self, x):
        x = self.fcs(x)
        x = self.fch(x)
        x = self.fce(x)
        return x

# Define the PINN model
class PINN(nn.Module):
    def __init__(self, N_INPUT=2, N_OUTPUT=1, N_HIDDEN=32, N_LAYERS=3):
        super(PINN, self).__init__()
        activation = nn.Tanh
        self.fcs = nn.Sequential(
            nn.Linear(N_INPUT, N_HIDDEN),
            activation()
        )
        self.hidden_layers = nn.ModuleList([
            nn.Sequential(
                nn.Linear(N_HIDDEN, N_HIDDEN),
                activation()
            ) for _ in range(N_LAYERS - 1)
        ])
        self.fce = nn.Linear(N_HIDDEN, N_OUTPUT)

    def forward(self, x):
        x = self.fcs(x)
        for layer in self.hidden_layers:
            x = layer(x)
        x = self.fce(x)
        return x

import torch.autograd as autograd
torch.manual_seed(123)

# Define the PDE residual
def pde_residual(model, x, t, a):
    x = x.requires_grad_()
    t = t.requires_grad_()
    u = model(torch.cat([x, t], dim=1))  # Neural network output
    u_t = autograd.grad(u.sum(), t, create_graph=True)[0]  # Time derivative
    u_x = autograd.grad(u.sum(), x, create_graph=True)[0]  # First spatial derivative
    u_xx = autograd.grad(u_x.sum(), x, create_graph=True)[0]  # Second spatial derivative
    return u_t - a**2 * u_xx  # Residual of the PDE
